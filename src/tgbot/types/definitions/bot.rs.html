150/// Represents a command of a bot.
151#[derive(Clone, Debug, Deserialize, Serialize)]
152pub struct BotCommand {
153    #[serde(rename = "command")]
154    name: String,
155    description: String,
156}
157
158impl BotCommand {
159    const MIN_NAME_LEN: usize = 1;
160    const MAX_NAME_LEN: usize = 32;
161    const MIN_DESCRIPTION_LEN: usize = 3;
162    const MAX_DESCRIPTION_LEN: usize = 256;
163
164    /// Creates a new `BotCommand`.
165    ///
166    /// # Arguments
167    ///
168    /// * `name` - The name of the command; 1-32 characters;
169    ///   can contain only lowercase English letters, digits and underscores.
170    /// * `description` - The description of the command; 3-256 characters.
171    pub fn new<C, D>(name: C, description: D) -> Result<Self, BotCommandError>
172    where
173        C: Into<String>,
174        D: Into<String>,
175    {
176        let name = name.into();
177        let description = description.into();
178        let name_len = name.len();
179        let description_len = description.len();
180        if !(Self::MIN_NAME_LEN..=Self::MAX_NAME_LEN).contains(&name_len) {
181            Err(BotCommandError::BadNameLen(name_len))
182        } else if !(Self::MIN_DESCRIPTION_LEN..=Self::MAX_DESCRIPTION_LEN).contains(&description_len) {
183            Err(BotCommandError::BadDescriptionLen(description_len))
184        } else {
185            Ok(Self { name, description })
186        }
187    }
188
189    /// Returns the name of the command.
190    pub fn name(&self) -> &str {
191        &self.name
192    }
193
194    /// Sets a new name for the command.
195    ///
196    /// # Arguments
197    ///
198    /// * `value` - The new name.
199    pub fn with_name<T>(mut self, value: T) -> Self
200    where
201        T: Into<String>,
202    {
203        self.name = value.into();
204        self
205    }
206
207    /// Returns the description of the command.
208    pub fn description(&self) -> &str {
209        &self.description
210    }
211
212    /// Sets a new description for the command.
213    ///
214    /// # Arguments
215    ///
216    /// * `value` - The new description.
217    pub fn with_description<T>(mut self, value: T) -> Self
218    where
219        T: Into<String>,
220    {
221        self.description = value.into();
222        self
223    }
224}

226/// Represents an error that can occur when creating a new [`BotCommand`].
227#[derive(Debug)]
228pub enum BotCommandError {
229    /// The provided name has an invalid length.
230    BadNameLen(usize),
231    /// The provided description has an invalid length.
232    BadDescriptionLen(usize),
233}
234
235impl Error for BotCommandError {}
236
237impl fmt::Display for BotCommandError {
238    fn fmt(&self, out: &mut fmt::Formatter) -> fmt::Result {
239        use self::BotCommandError::*;
240        match self {
241            BadNameLen(len) => write!(
242                out,
243                "command name can have a length of {} up to {} characters, got {}",
244                BotCommand::MIN_NAME_LEN,
245                BotCommand::MAX_NAME_LEN,
246                len
247            ),
248            BadDescriptionLen(len) => write!(
249                out,
250                "command description can have a length of {} up to {} characters, got {}",
251                BotCommand::MIN_DESCRIPTION_LEN,
252                BotCommand::MAX_DESCRIPTION_LEN,
253                len
254            ),
255        }
256    }
257}

259/// Represents a scope to which bot commands are applied.
260#[derive(Clone, Debug, Deserialize, PartialEq, PartialOrd, Serialize)]
261#[serde(tag = "type")]
262#[serde(rename_all = "snake_case")]
263pub enum BotCommandScope {
264    /// All group and supergroup chat administrators.
265    AllChatAdministrators,
266    /// All group and supergroup chats.
267    AllGroupChats,
268    /// All private chats.
269    AllPrivateChats,
270    /// A specific chat.
271    Chat {
272        /// Unique identifier of the target chat.
273        chat_id: ChatId,
274    },
275    /// All administrators of a specific group or supergroup chat.
276    ChatAdministrators {
277        /// Unique identifier of the target chat.
278        chat_id: ChatId,
279    },
280    /// A specific member of a group or supergroup chat.
281    ChatMember {
282        /// Unique identifier of the target chat.
283        chat_id: ChatId,
284        /// Unique identifier of the target user.
285        user_id: Integer,
286    },
287    /// Default scope.
288    ///
289    /// Default commands are used if no commands with a narrower scope are specified for a user.
290    Default,
291}
292
293impl BotCommandScope {
294    /// Creates a new `BotCommandScope` covering a specific chat.
295    ///
296    /// # Arguments
297    ///
298    /// * `value` - Chat ID.
299    pub fn chat<T>(value: T) -> Self
300    where
301        T: Into<ChatId>,
302    {
303        Self::Chat { chat_id: value.into() }
304    }
305
306    /// Creates a new `BotCommandScope` covering all administrators
307    /// of a specific group or supergroup chat.
308    ///
309    /// # Arguments
310    ///
311    /// * `value` - Chat ID.
312    pub fn chat_administrators<T>(value: T) -> Self
313    where
314        T: Into<ChatId>,
315    {
316        Self::ChatAdministrators { chat_id: value.into() }
317    }
318
319    /// Creates a new `BotCommandScope` covering a specific member of a group or supergroup chat.
320    ///
321    /// # Arguments
322    ///
323    /// * `chat_id` - Chat ID.
324    /// * `user_id` - User ID.
325    pub fn chat_member<A>(chat_id: A, user_id: Integer) -> Self
326    where
327        A: Into<ChatId>,
328    {
329        Self::ChatMember {
330            chat_id: chat_id.into(),
331            user_id,
332        }
333    }
334}
