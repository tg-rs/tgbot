33/// Describes a message that can be inaccessible to the bot.
34#[derive(Clone, Debug, derive_more::From, PartialEq, Serialize)]
35#[serde(untagged)]
36pub enum MaybeInaccessibleMessage {
37    /// Describes a message that was deleted or is otherwise inaccessible to the bot.
38    InaccessibleMessage(InaccessibleMessage),
39    /// Describes a regular message.
40    Message(Box<Message>),
41}
42
43impl<'de> Deserialize<'de> for MaybeInaccessibleMessage {
44    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
45    where
46        D: Deserializer<'de>,
47    {
48        Message::deserialize(deserializer).map(|x| {
49            if x.date == 0 {
50                Self::InaccessibleMessage(InaccessibleMessage {
51                    chat: x.chat,
52                    message_id: x.id,
53                })
54            } else {
55                Self::Message(Box::new(x))
56            }
57        })
58    }
59}

82    /// Sender of the message.
83    #[serde(flatten)]

87    /// Unique identifier of the business connection from which the message was received.

89    /// If non-empty, the message belongs to a chat of the corresponding business account

91    pub business_connection_id: Option<String>,
92    /// Information about the direct messages chat topic that contains the message.
93    pub direct_messages_topic: Option<DirectMessagesTopic>,
94    /// Unique identifier of the message effect added to the message.
95    pub effect_id: Option<String>,
96    /// Information about the message that is being replied to, which may come from another chat or forum topic.
97    pub external_reply: Option<ExternalReplyInfo>,
98    /// formation about the original message for forwarded messages.
99    pub forward_origin: Option<MessageOrigin>,
100    /// Indicates whether the message media is covered by a spoiler animation.
101    pub has_media_spoiler: Option<bool>,
102    /// Whether the message was sent by an implicit action.
103    ///
104    /// For example, as an away or a greeting business message, or as a scheduled message.
105    pub is_from_offline: Option<bool>,
106    /// Whether the message is a paid post.
107    ///
108    /// Note that such posts must not be deleted for 24 hours to receive the payment and can't be edited.
109    pub is_paid_post: Option<bool>,
110    /// Indicates whether the message is sent to a topic in a forum supergroup or a private chat with the bot.
111    pub is_topic_message: Option<bool>,
112    /// Options used for link preview generation for the message,
113    /// if it is a text message and link preview options were changed.
114    pub link_preview_options: Option<LinkPreviewOptions>,
115    /// Unique identifier of a media message group this message belongs to.
116    pub media_group_id: Option<String>,
117    /// Unique identifier of a message thread to which the message belongs;
118    /// for supergroups and private chats only.
119    pub message_thread_id: Option<Integer>,
120    /// The number of Telegram Stars that were paid by the sender of the message to send it.
121    pub paid_star_count: Option<Integer>,
122    /// For replies that quote part of the original message, the quoted part of the message.
123    pub quote: Option<TextQuote>,
124    /// Inline keyboard attached to the message.
125    pub reply_markup: Option<InlineKeyboardMarkup>,
126    /// For replies, the original message or story.
127    #[serde(flatten)]
128    pub reply_to: Option<ReplyTo>,
129    /// Identifier of the specific checklist task that is being replied to.
130    pub reply_to_checklist_task_id: Option<Integer>,
131    /// Number of boosts added by the user.
132    ///
133    /// Contains a value only if the sender of the message boosted the chat.
134    pub sender_boost_count: Option<Integer>,
135    /// The bot that actually sent the message on behalf of the business account.
136    ///
137    /// Available only for outgoing messages sent on behalf of the connected business account.
138    pub sender_business_bot: Option<User>,
139    /// Whether the caption must be shown above the message media.
140    pub show_caption_above_media: Option<bool>,
141    /// Information about suggested post parameters if the message is a suggested post in a channel direct messages chat.
142    ///
143    /// If the message is an approved or declined suggested post, then it can't be edited.
144    pub suggested_post_info: Option<SuggestedPostInfo>,
145    /// Bot through which the message was sent.
146    pub via_bot: Option<User>,
147
148    /// Contains message data.
149    #[serde(flatten)]
150    pub data: MessageData,
151}

208    /// Returns a text of the message (includes caption).
209    pub fn get_text(&self) -> Option<&Text> {
210        match &self.data {
211            MessageData::Text(text) => Some(text),
212            MessageData::Audio(audio) => audio.caption.as_ref(),
213            MessageData::Document(doc) => doc.caption.as_ref(),
214            MessageData::Photo(photo) => photo.caption.as_ref(),
215            MessageData::Video(video) => video.caption.as_ref(),
216            MessageData::Voice(voice) => voice.caption.as_ref(),
217            _ => None,
218        }
219    }
